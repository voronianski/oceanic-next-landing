import React from 'react';

class MyComponent extends React.Component {
    constructor() {
        super();

        this.state = {
            title: 'World'
        };
    }

    clickHandler(name) {
        this.setState(name);
    }

    render() {
        let { title } = this.state;

        return (
            <div>
                <h1>Hello, {title}!</h1>
                <button onClick={() => this.clickHandler('React')}>
                    Change title
                </button>
            </div>
        );
    }
}

export default MyComponent;
